			NavigationActions.reset({
				index:0,
				actions:[
					0：NavigationActions.navigate({routeName:"xx"}),
					1：NavigationActions.navigate({routeName:"OO"}),
					2：NavigationActions.navigate({routeName:"XXOO"})
				]
				
				"XX"  "OO"  是之前路由中已经定义的
